#include <Keyboard.h>

boolean risingEdge;
float frameLength = 1000.0/60.0;

int UP = 119;
int DOWN = 115;
int LEFT = 97;
int RIGHT = 100;
int A = 117;
int B = 105;
int C = 111;
int D = 32;
int E = 106;

void setup() {
	pinMode(1, INPUT_PULLUP);
	Keyboard.begin();
	risingEdge = false;
}

void loop() {
	if(!digitalRead(1) && !risingEdge) {
		risingEdge = true;
		combo();
	} else if(digitalRead(1)){
		risingEdge = false;
	}
}

void combo() {
	Keyboard.press(DOWN);
	Keyboard.press(A);
	delay((1.0) * frameLength + 2);
	Keyboard.release(A);
	Keyboard.release(DOWN);
	delay(7.5 * frameLength + 2);
	Keyboard.press(B);
	delay((1.0 + 7.5) * frameLength + 2);
	Keyboard.release(B);
	delay(8.0 * frameLength + 2);
	Keyboard.press(DOWN);
	Keyboard.press(B);
	delay((1.0 + 8.0) * frameLength + 2);
	Keyboard.release(B);
	delay(9.0 * frameLength + 2);
	Keyboard.press(DOWN);
	Keyboard.press(C);
	delay((1.0 + 9.0) * frameLength + 2);
	Keyboard.release(C);
	delay(9.5 * frameLength + 2);
	Keyboard.press(DOWN);
	Keyboard.press(C);
	Keyboard.press(RIGHT);
	delay((1.0 + 9.5) * frameLength + 2);
	Keyboard.release(C);
	Keyboard.release(RIGHT);
	Keyboard.release(DOWN);
	delay(10.0 * frameLength + 2);
	Keyboard.press(UP);
	Keyboard.press(RIGHT);
	delay((1.0 + 10.0) * frameLength + 2);
	Keyboard.release(UP);
	Keyboard.release(RIGHT);
	Keyboard.release(DOWN);
	delay(0.5 * frameLength + 2);
	Keyboard.press(B);
	delay((1.0 + 0.5) * frameLength + 2);
	Keyboard.release(B);
	Keyboard.release(DOWN);
	delay(9.0 * frameLength + 2);
	Keyboard.press(C);
	delay((1.0 + 9.0) * frameLength + 2);
	Keyboard.release(C);
	Keyboard.release(DOWN);
	delay(10.5 * frameLength + 2);
	Keyboard.press(UP);
	Keyboard.press(RIGHT);
	delay((1.0 + 10.5) * frameLength + 2);
	Keyboard.release(UP);
	Keyboard.release(RIGHT);
	Keyboard.release(DOWN);
	delay(0.5 * frameLength + 2);
	Keyboard.press(B);
	delay((1.0 + 0.5) * frameLength + 2);
	Keyboard.release(B);
	Keyboard.release(DOWN);
	delay(9.0 * frameLength + 2);
	Keyboard.press(C);
	delay((1.0 + 9.0) * frameLength + 2);
	Keyboard.release(C);
	Keyboard.release(DOWN);
	delay(10.5 * frameLength + 2);
	Keyboard.press(E);
	Keyboard.press(RIGHT);
	delay((1.0 + 10.5) * frameLength + 2);
	Keyboard.release(E);
	Keyboard.release(RIGHT);
}
