#include <Keyboard.h>

boolean risingEdge;
float frameLength = 1000.0/60.0;

int UP = 119;
int DOWN = 115;
int LEFT = 97;
int RIGHT = 100;
int A = 117;
int B = 105;
int C = 111;
int D = 32;
int E = 106;

void setup() {
	pinMode(1, INPUT_PULLUP);
	Keyboard.begin();
	risingEdge = false;
}

void loop() {
	if(!digitalRead(1) && !risingEdge) {
		risingEdge = true;
		combo();
	} else if(digitalRead(1)){
		risingEdge = false;
	}
}

void combo() {
	Keyboard.press(DOWN);
	Keyboard.press(A);
	delay(4 * frameLength);
	Keyboard.release(A);
	Keyboard.release(DOWN);
	delay(7.0 * frameLength);
	Keyboard.press(B);
	delay(7 * frameLength);
	Keyboard.release(B);
	delay(9.0 * frameLength);
	Keyboard.press(DOWN);
	Keyboard.press(B);
	delay(6 * frameLength);
	Keyboard.release(B);
	delay(9.5 * frameLength);
	Keyboard.press(DOWN);
	Keyboard.press(C);
	delay(7 * frameLength);
	Keyboard.release(C);
	Keyboard.release(DOWN);
	delay(10.5 * frameLength);
	Keyboard.press(C);
	delay(4 * frameLength);
	Keyboard.release(C);
	Keyboard.release(DOWN);
	delay(10.5 * frameLength);
	Keyboard.press(UP);
	delay(4 * frameLength);
	Keyboard.release(UP);
	Keyboard.release(DOWN);
	delay(0.5 * frameLength);
	Keyboard.press(A);
	delay(5 * frameLength);
	Keyboard.release(A);
	Keyboard.release(DOWN);
	delay(7.5 * frameLength);
	Keyboard.press(B);
	delay(4 * frameLength);
	Keyboard.release(B);
	Keyboard.release(DOWN);
	delay(12.0 * frameLength);
	Keyboard.press(C);
	delay(5 * frameLength);
	Keyboard.release(C);
}
